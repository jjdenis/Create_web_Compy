sc.printf 

 

    # Simple print
    sc.printf('Hello')

    # Print on a different color
    sc.printf('Hello', color='black')

    # Colors can be chosen by numbers
    sc.printf('Hello', color=4)

    sc.printf('Hello, ', stay=True)

    sc.printf('partner!', color=0)

    sc.printf()

    sc.printf('Hello', reverse = True)

    sc.printf('Hello, or better...', color='light-green', reverse=True, stay=True)

    sc.printf('Good bye!!!')
